def entity_exists? artist
    return Artist.exists?(artist.id) unless artist.id.nil?
    return Artist.exists?(mbid: artist.mbid) unless artist.mbid.nil?
    return Artist.exists?(name: artist.name) unless artist.name.nil?
    return Artist.exists?(slug: artist.slug) unless artist.slug.nil?
    false
end

def populate_entity artist

    dbArtist = nil
    dbArtist = Artist.find_by_mbid(artist.mbid) unless artist.mbid.nil? or !dbArtist.nil?
    dbArtist = Artist.find_by_name(artist.name) unless artist.name.nil? or !dbArtist.nil?
    dbArtist = Artist.find_by_slug(artist.slug) unless artist.slug.nil? or !dbArtist.nil?

    # When we have loaded a matching artist using the unique keys,

    # ensure

    unless dbArtist.nil? or !dbArtist.is_a? Artist
        artist.assign_attributes(dbArtist.attributes)
    end

    artist
end

def preload_set entities
    entities.each_with_index do |entity, idx|
        unless entity_exists?(entity)
            entity.save!
        else
            entities[idx] = populate_entity entity
        end
    end
end
