def self.generate_artist_entity data
    if !data["mbid"].empty?
        artist = Artist.new
        artist.name = data["name"]
        artist.mbid = data["mbid"]
        artist.thumbnail_source = data["image"][3]['#text']
        artist
    end
end

def self.generate_top_artists_datalist data
    list = Array.new
    data["artists"]["artist"].each do |artistData|
        list << generate_artist_entity(artistData)
    end
    list
end
