def self.create_album_list artist, release_groups
    if artist.albums.nil?
        artist.albums = Array.new
    end

    release_groups.each do |release|
        album = Album.new
        album.title = release.title
        album.mbid = release.id
        artist.albums << album
    end

    Album.preload_set artist.albums
end

def self.find_remote artist
    return MusicBrainz::Artist.find(artist.mbid) unless artist.mbid.nil?
    return MusicBrainz::Artist.find_by_name(artist.name) unless artist.name.nil?
end
